###Abundance
conda activate bowtie2

bowtie2-build All_MAG_cat.fa All_MAG_cat.fa

bowtie2 -x All_MAG_cat.fa -1 clean_1.fastq.gz -2 clean_2.fastq.gz -S out.sam --threads 48

samtools view -@ 14 -b -S out.sam > out.bam

samtools sort -@ 14 out.bam > out.sorted.bam

coverm genome -m relative_abundance --min-read-percent-identity 0.90 --bam-files out.sorted.bam --genome-fasta-directory MAG/ > coverm_abundance_${i}.txt


coverm genome \
--genome-fasta-directory MAG/ \
-m rpkm \
--bam-files out.sorted.bam \
--min-read-percent-identity 95 \
--min-read-aligned-percent 90 \
--output-file abundance_rpkm.tsv \
--contig-end-exclusion 0 \
--no-zeros \
-t 42
